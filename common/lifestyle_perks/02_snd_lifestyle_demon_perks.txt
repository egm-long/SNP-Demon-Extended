# Minor Demon
minor_demon_perk = {
	lifestyle = demon_lifestyle
	tree = magic
	position = { 1 0 }
	icon = node_vampire # Anpassungen Notwendig 

	can_be_picked = {
		has_demon_trigger = yes
	}
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_demon_interest_trigger = yes
			}
			add = 3000
		}
		if = {
			limit = {
				has_focus = demon_demon_supremacy_focus
			}
			multiply = 5
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
	}
	
	character_modifier = {
        intrigue = 3
		#monthly_income = 2
		add source_xp = 2
	}
	
	
}
