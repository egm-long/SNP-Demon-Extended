rumors_of_demons = {
    icon = dread
    is_valid = {
        is_current_phase_active = yes
        any_attending_character = {
            NOT = { this = scope:host }
            location = scope:province
        }
    }

	weight = {
		value = 1
		if = {
			limit = {
				scope:host = {
					has_trait = paranoid
				}
			}
			add = 0.5
		}
	}


    effect = {
        scope:host = {
            save_scope_as = first
        }

        add_activity_log_entry = {
            key = rumors_of_demons
            tags = { pulse_action }
            character = scope:first

            scope:first = {
                add_stress = minor_stress_gain
            }
        }
    }
}

expertise_display_demon = {
    icon = public
    is_valid = {
        is_current_phase_active = yes
        any_attending_character = {
            NOT = { this = scope:host }
            location = scope:province
        }
    }

	weight = {
		value = 1
		if = {
			limit = {
				scope:host = {
					prowess > 20
				}
			}
			add = 0.5
		}
	}


    effect = {
        random_attending_character = {
            limit = {
                NOT = { this = scope:host }
                location = scope:province
            }
            save_scope_as = first
        }

		if = {
			limit = {
				scope:host = {
					has_trait = demon
				}
			}
			add_activity_log_entry = {
				key = expertise_demon_display
				tags = { pulse_action }
				character = scope:first

				scope:first = {
					add_opinion = {
						target = scope:root
						modifier = impressed_opinion
						opinion = 20
					}
				}
			}
		}
		else = {
			limit = {
				scope:host = {
					has_trait = demon
				}
			}
			add_activity_log_entry = {
				key = expertise_hybrid_display
				tags = { pulse_action }
				character = scope:first

				scope:first = {
					add_opinion = {
						target = scope:root
						modifier = impressed_opinion
						opinion = 20
					}
				}
			}
		}
		
    }
}
